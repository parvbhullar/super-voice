Insert the security check right after the transaction is received and logged (after line ~233 `info!(?key, "received transaction")`), before the to-tag check at line ~234:

```rust
// -- SIP Security Check --
// Enforce IP blacklist, flood protection, brute-force protection,
// and UA blacklist on every inbound SIP message BEFORE any routing.
if let Some(ref security_module) = self.security_module {
    // Extract source IP from the connection (remote peer address)
    let source_ip = tx.connection.as_ref()
        .and_then(|conn| {
            let addr = conn.get_addr();
            // get_addr() returns a SocketAddr-like type; extract the IP portion
            Some(addr.ip().to_string())
        });

    // Extract User-Agent header from the SIP message
    let user_agent = tx.original.headers.iter().find_map(|h| {
        match h {
            rsip::Header::Other(name, value) if name.eq_ignore_ascii_case("User-Agent") => {
                Some(value.as_str())
            }
            _ => None,
        }
    });

    if let Some(ref ip_str) = source_ip {
        let result = security_module.check_request(ip_str, user_agent);
        match result {
            SecurityCheckResult::Allowed | SecurityCheckResult::Whitelisted => {
                // Pass through — whitelisted or no block
            }
            SecurityCheckResult::Blacklisted => {
                warn!(source_ip = %ip_str, "SIP security: blacklisted IP — dropping");
                // Silent drop — do not reply to blacklisted IPs (avoid fingerprinting)
                continue;
            }
            SecurityCheckResult::FloodBlocked { count } => {
                warn!(source_ip = %ip_str, count, "SIP security: flood-blocked IP — rejecting 503");
                let _ = tx.reply_with(
                    rsip::StatusCode::ServiceUnavailable,
                    vec![rsip::Header::Other(
                        "Reason".into(),
                        "SIP;cause=503;text=\"Rate limit exceeded\"".into(),
                    )],
                    None,
                ).await;
                continue;
            }
            SecurityCheckResult::BruteForceBlocked { failures } => {
                warn!(source_ip = %ip_str, failures, "SIP security: brute-force-blocked IP — rejecting 403");
                let _ = tx.reply_with(
                    rsip::StatusCode::Forbidden,
                    vec![rsip::Header::Other(
                        "Reason".into(),
                        "SIP;cause=403;text=\"Too many auth failures\"".into(),
                    )],
                    None,
                ).await;
                continue;
            }
            SecurityCheckResult::UaBlocked { ref ua } => {
                warn!(source_ip = %ip_str, ua = %ua, "SIP security: blocked user-agent — dropping");
                // Silent drop for scanner UAs
                continue;
            }
            SecurityCheckResult::InvalidMessage { ref reason } => {
                warn!(source_ip = %ip_str, reason = %reason, "SIP security: invalid message — rejecting 400");
                let _ = tx.reply(rsip::StatusCode::BadRequest).await;
                continue;
            }
        }
    }
}
```
