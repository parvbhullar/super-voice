3. **scripts/check_startup.sh** — Concrete startup time validation script:
   ```bash
   #!/usr/bin/env bash
   set -euo pipefail
   # Validate binary startup time is under 1 second.
   # Usage: ./scripts/check_startup.sh [path-to-binary]
   BINARY="${1:-target/release/active-call}"
   if [ ! -f "$BINARY" ]; then
     echo "ERROR: Binary not found at $BINARY"
     exit 1
   fi
   START=$(date +%s%N)
   timeout 5 "$BINARY" --help >/dev/null 2>&1 || true
   END=$(date +%s%N)
   ELAPSED_MS=$(( (END - START) / 1000000 ))
   echo "Startup time: ${ELAPSED_MS}ms"
   if [ "$ELAPSED_MS" -gt 1000 ]; then
     echo "FAIL: Startup took ${ELAPSED_MS}ms (limit: 1000ms)"
     exit 1
   fi
   echo "PASS: Startup under 1 second"
   ```
